1. sudo npm install yarn --global
    
    ## To install the Yarn package manager, run:
     curl -sL https://dl.yarnpkg.com/debian/pubkey.gpg | sudo apt-key add -
     echo "deb https://dl.yarnpkg.com/debian/ stable main" | sudo tee /etc/apt/sources.list.d/yarn.list
     sudo apt-get update && sudo apt-get install yarn
     yarn --version

2. Install live-server
  sudo yarn global add live-server
  or
  sudo npm install -g live-server
  live-server -v

3. run life-server 
  live-server folder_name  
    for example
  live-server public

4. Install Babel
    sudo yarn global add babel-cli@6.24.1
    yarn init
    // install env preset  
    // install react preset 
    yarn add babel-preset-react@6.24.1 babel-preset-env@1.5.2

5. Babel config    
    to generate js from JSX
  yarn install  // install all dependencies
  babel src/app.js --out-file=public/scripts/app.js --presets=env,react
    for watch changes
   babel src/app.js --out-file=public/scripts/app.js --presets=env,react --watch

6. Webpack
  yarn add webpack@3.1.0  
  then create 'webpack.config.js' in root project folder
    basis config
      const PATH = require('path');
      module.exports = {
        entry : './src/app.js',
        output : {
          path : PATH.join(__dirname, 'public'),
          filename : 'bundle.js'
        },
      };
